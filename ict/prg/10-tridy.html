📖 Co je třída a proč ji používáme?

Třída je šablona (blueprint) pro vytváření objektů. Je základem objektově orientovaného programování (OOP). Třída definuje:

  • Atributy (data): Vlastnosti, které objekt má (jméno, věk, barva...)
  • Metody (chování): Co objekt umí dělat (chodit, mluvit, počítat...)

💡 Analogie ze života:

  • Třída = Architektonický plán domu
  • Objekt = Konkrétní postavený dům podle plánu
  • Z jednoho plánu můžeš postavit mnoho domů, každý s jiným číslem popisným

Třída v C++

class Osoba {
private:                          // Soukromé členy - nelze přistoupit zvenčí
    string jmeno;
    int vek;

public:                           // Veřejné členy - přístupné odkudkoli
    // Konstruktor - volá se při vytvoření objektu
    Osoba(string j, int v) {
        jmeno = j;
        vek = v;
        cout << "Osoba " << jmeno << " vytvořena!" << endl;
    }
    
    // Destruktor - volá se při zániku objektu
    ~Osoba() {
        cout << "Osoba " << jmeno << " zaniká!" << endl;
    }
    
    // Metody (členské funkce)
    void pozdrav() {
        cout << "Ahoj, jsem " << jmeno << " a mám " << vek << " let." << endl;
    }
    
    // Getter - vrací hodnotu privátní proměnné
    int getVek() {
        return vek;
    }
    
    // Setter - nastavuje hodnotu s kontrolou
    void setVek(int novy_vek) {
        if (novy_vek > 0 && novy_vek < 150) {
            vek = novy_vek;
        }
    }
}; // Nezapomeň na středník!

Třída v C#

class Osoba {
    // Privátní pole (backing fields)
    private string jmeno;
    private int vek;
    
    // Konstruktor
    public Osoba(string j, int v) {
        jmeno = j;
        vek = v;
        Console.WriteLine($"Osoba {jmeno} vytvořena!");
    }
    
    // Property (vlastnost) - C# elegantní způsob getterů/setterů
    public int Vek {
        get { return vek; }
        set { 
            if (value > 0 && value < 150)
                vek = value; 
        }
    }
    
    // Auto-property (zkrácená syntaxe)
    public string Email { get; set; }
    
    // Read-only property
    public string Info => $"{jmeno}, {vek} let";
    
    // Metoda
    public void Pozdrav() {
        Console.WriteLine($"Ahoj, jsem {jmeno}!");
    }
}

Přístupové modifikátory

Určují, odkud lze k členům třídy přistupovat. Základní princip zapouzdření.

Modifikátor Přístup Použití
public Odkudkoli Metody, veřejné API
private Jen v dané třídě Interní data, pomocné metody
protected V třídě a potomcích Sdílená data pro dědičnost
internal (C#) V rámci assembly Mezi třídami v projektu

💡 Pravidlo palce: Data (proměnné) dělej private, metody podle potřeby public nebo private.

Konstruktory

Speciální metoda volaná automaticky při vytvoření objektu. Slouží k inicializaci. 

class Auto {
    string znacka;
    int rokVyroby;
    
public:
    // Výchozí konstruktor (bez parametrů)
    Auto() {
        znacka = "Neznámá";
        rokVyroby = 2020;
    }
    
    // Parametrický konstruktor
    Auto(string z, int r) {
        znacka = z;
        rokVyroby = r;
    }
    
    // C++: Konstruktor s inicializačním seznamem (preferovaný)
    Auto(string z, int r) : znacka(z), rokVyroby(r) { }
};

// Vytvoření objektů
Auto auto1;                    // Volá výchozí konstruktor
Auto auto2("Škoda", 2022);    // Volá parametrický konstruktor

Přetěžování konstruktorů

Třída může mít více konstruktorů s různými parametry:

// C#
class Osoba {
    public string Jmeno { get; set; }
    public int Vek { get; set; }
    
    public Osoba() {                           // Výchozí
        Jmeno = "Anonym";
        Vek = 0;
    }
    
    public Osoba(string jmeno) {               // S jedním parametrem
        Jmeno = jmeno;
        Vek = 0;
    }
    
    public Osoba(string jmeno, int vek) {      // Se dvěma parametry
        Jmeno = jmeno;
        Vek = vek;
    }
}

Destruktory

Metoda volaná při zániku objektu. Slouží k úklidu (uvolnění paměti, zavření souborů).

// C++ - Destruktor je DŮLEŽITÝ pro správu paměti!
class Soubor {
    FILE* file;
public:
    Soubor(const char* nazev) {
        file = fopen(nazev, "r");
    }
    
    ~Soubor() {                    // Destruktor: ~NázevTřídy()
        if (file) fclose(file);    // Uvolníme prostředky
        cout << "Soubor zavřen" << endl;
    }
};
// Destruktor se volá automaticky při opuštění scope nebo delete

// C# - Garbage Collector se stará o paměť automaticky
// Destruktor (finalizér) se používá zřídka
~Osoba() {
    Console.WriteLine("Finalizace");
}
// Preferuj IDisposable a using pro deterministické uvolňování

Vytvoření instance (objektu)

// C++ - dvě možnosti
Osoba jan("Jan", 25);              // Na zásobníku (stack) - automatický destruktor
Osoba* eva = new Osoba("Eva", 30); // Na haldě (heap) - musíš smazat ručně!
delete eva;                         // Uvolnění paměti

// C# - vždy new (objekty jsou na haldě, GC uklízí)
Osoba jan = new Osoba("Jan", 25);
// Nepotřebuješ delete, Garbage Collector se postará

Zapouzdření (Encapsulation)

Jeden z pilířů OOP. Skrytí vnitřní implementace a přístup přes veřejné rozhraní.

class BankovniUcet {
private:
    double zustatek;        // Skryto - nelze přímo měnit
    
public:
    BankovniUcet(double pocatecni) {
        if (pocatecni >= 0) zustatek = pocatecni;
    }
    
    // Veřejné rozhraní s kontrolou
    void vloz(double castka) {
        if (castka > 0) {
            zustatek += castka;
        }
    }
    
    bool vyber(double castka) {
        if (castka > 0 && castka <= zustatek) {
            zustatek -= castka;
            return true;
        }
        return false;  // Nedostatek prostředků
    }
    
    double getZustatek() {
        return zustatek;
    }
};

// Použití:
BankovniUcet ucet(1000);
ucet.vloz(500);              // OK
ucet.vyber(200);             // OK
ucet.zustatek = 1000000;     // CHYBA! Private člen

💡 Proč zapouzdření?

  • Ochrana dat před neplatnými hodnotami
  • Můžeš změnit implementaci bez změny rozhraní
  • Snazší testování a ladění

🎓 Shrnutí pro maturitu

  • Třída = šablona pro objekty (atributy + metody)
  • Objekt/Instance = konkrétní výtvor podle třídy
  • Konstruktor = inicializace při vytváření (stejný název jako třída)
  • Destruktor (C++) = úklid při zániku (~NázevTřídy)
  • public/private/protected = řízení přístupu
  • Zapouzdření = private data + public metody
  • C++: Osoba jan("Jan", 25); nebo new/delete
  • C#: Osoba jan = new Osoba("Jan", 25); + Properties
